186 {

188 
189  if (jobIdx >= kp.njobs)
190  return;
191 
192  float2 pos = make_float2(initial[jobIdx].x, initial[jobIdx].y);
193  float mean = kp.imgmeans[jobIdx];
194  float I0 = mean*0.5f*kp.images.w*kp.images.h;
195  float bg = mean*0.5f;
196 
197  const float _1oSq2Sigma = 1.0f / (sqrtf(2) * sigma);
198  const float _1oSq2PiSigma = (1.0f / (sqrtf(2*3.14159265359f))) / sigma;
199  const float _1oSq2PiSigma3 = (1.0f / (sqrtf(2*3.14159265359f))) / (sigma*sigma*sigma);
200 
201  for (int i=0;i<iterations;i++)
202  {
203  float dL_dx = 0.0;
204  float dL_dy = 0.0;
205  float dL_dI0 = 0.0;
206  float dL_dIbg = 0.0;
207  float dL2_dx = 0.0;
208  float dL2_dy = 0.0;
209  float dL2_dI0 = 0.0;
210  float dL2_dIbg = 0.0;
211 
212  for (int y=0;y<kp.images.h;y++)
213  {
214  for (int x=0;x<kp.images.w;x++)
215  {
216  float Xexp0 = (x-pos.x + .5f) * _1oSq2Sigma;
217  float Yexp0 = (y-pos.y + .5f) * _1oSq2Sigma;
218 
219  float Xexp1 = (x-pos.x - .5f) * _1oSq2Sigma;
220  float Yexp1 = (y-pos.y - .5f) * _1oSq2Sigma;
221 
222  float DeltaX = 0.5f * erff(Xexp0) - 0.5f * erff(Xexp1);
223  float DeltaY = 0.5f * erff(Yexp0) - 0.5f * erff(Yexp1);
224  float mu = bg + I0 * DeltaX * DeltaY;
225 
226  float dmu_dx = I0*_1oSq2PiSigma * ( expf(-Xexp1*Xexp1) - expf(-Xexp0*Xexp0)) * DeltaY;
227 
228  float dmu_dy = I0*_1oSq2PiSigma * ( expf(-Yexp1*Yexp1) - expf(-Yexp0*Yexp0)) * DeltaX;
229  float dmu_dI0 = DeltaX*DeltaY;
230  float dmu_dIbg = 1;
231 
232  float smp = TImageSampler::Index(kp.images, x,y, jobIdx);
233  float f = smp / mu - 1;
234  dL_dx += dmu_dx * f;
235  dL_dy += dmu_dy * f;
236  dL_dI0 += dmu_dI0 * f;
237  dL_dIbg += dmu_dIbg * f;
238 
239  float d2mu_dx = I0*_1oSq2PiSigma3 * ( (x - pos.x - .5f) * expf (-Xexp1*Xexp1) - (x - pos.x + .5) * expf(-Xexp0*Xexp0) ) * DeltaY;
240  float d2mu_dy = I0*_1oSq2PiSigma3 * ( (y - pos.y - .5f) * expf (-Yexp1*Yexp1) - (y - pos.y + .5) * expf(-Yexp0*Yexp0) ) * DeltaX;
241  dL2_dx += d2mu_dx * f - dmu_dx*dmu_dx * smp / (mu*mu);
242  dL2_dy += d2mu_dy * f - dmu_dy*dmu_dy * smp / (mu*mu);
243  dL2_dI0 += -dmu_dI0*dmu_dI0 * smp / (mu*mu);
244  dL2_dIbg += -smp / (mu*mu);
245  }
246  }
247 
248  pos.x -= dL_dx / dL2_dx;
249  pos.y -= dL_dy / dL2_dy;
250  I0 -= dL_dI0 / dL2_dI0;
251  bg -= dL_dIbg / dL2_dIbg;
252  }
253 
254 
255  positions[jobIdx].x = pos.x;
256  positions[jobIdx].y = pos.y;
257  if (I_bg) I_bg[jobIdx] = bg;
258  if (I_0) I_0[jobIdx] = I0;
259 }
