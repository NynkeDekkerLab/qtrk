160  static CFFT_BOTH void kf_work( int stage,cpx_type * Fout, const cpx_type * f, size_t fstride,size_t in_stride, KernelParams* kparm)
161  {
162  int p = kparm->radix(stage);
163  int m = kparm->remainder(stage);
164  cpx_type * Fout_beg = Fout;
165  cpx_type * Fout_end = Fout + p*m;
166 
167  if (m==1) {
168  do{
169  *Fout = *f;
170  f += fstride*in_stride;
171  }while(++Fout != Fout_end );
172  }else{
173  do{
174  // recursive call:
175  // DFT of size m*p performed by doing
176  // p instances of smaller DFTs of size m,
177  // each one takes a decimated version of the input
178  kf_work(stage+1, Fout , f, fstride*p,in_stride,kparm);
179  f += fstride*in_stride;
180  }while( (Fout += m) != Fout_end );
181  }
182 
183  Fout=Fout_beg;
184 
185  // recombine the p smaller DFTs
186  switch (p) {
187  case 2:
188  kf_bfly2(Fout,fstride,m, kparm);
189  break;
190  case 3: kf_bfly3(Fout,fstride,m, kparm); break;
191  case 4: kf_bfly4(Fout,fstride,m, kparm); break;
192  case 5: kf_bfly5(Fout,fstride,m, kparm); break;
193  default: kf_bfly_generic(Fout,fstride,m,p,kparm); break;
194  }
195  }
